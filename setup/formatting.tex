% ----------------------------------------------------------------------------
% PAGE LAYOUT

\usepackage[a4paper, lmargin=0.15\paperwidth, rmargin=0.15\paperwidth, tmargin=0.1111\paperheight, bmargin=0.1111\paperheight]{geometry}

% ----------------------------------------------------------------------------
% TYPOGRAPHY SETTINGS
% ----------------------------------------------------------------------------
\frenchspacing
\linespread{1.2}
\usepackage[parfill]{parskip}

% ----------------------------------------------------------------------------
% HEADERS AND FOOTERS
% ----------------------------------------------------------------------------
\setlength{\headheight}{13.6pt}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\small\leftmark}
\fancyhead[R]{\small\rightmark}
\fancyfoot[L]{\footnotesize\footerLeft}
\fancyfoot[C]{\footnotesize\thepage}
\fancyfoot[R]{\footnotesize\footerRight}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0.4pt}
\renewcommand{\chaptermark}[1]{\markboth{#1}{}}
\renewcommand{\sectionmark}[1]{\markright{\thesection\ #1}}

% Clear headers on title page and table of contents
\fancypagestyle{plain}{%
    \fancyhf{}
    \fancyfoot[L]{\footnotesize\footerLeft}
    \fancyfoot[C]{\footnotesize\thepage}
    \fancyfoot[R]{\footnotesize\footerRight}
    \renewcommand{\headrulewidth}{0pt}
    \renewcommand{\footrulewidth}{0.4pt}
}

% ----------------------------------------------------------------------------
% CHAPTER FORMATTING
% ----------------------------------------------------------------------------
\makeatletter
\def\@makechapterhead#1{%
  \vspace*{1\baselineskip}%
  {\parindent\z@ \raggedright\normalfont
   \Huge\bfseries #1\par
   \nobreak
   \vspace{0.5\baselineskip}}}
\def\@makeschapterhead#1{%
  \vspace*{1\baselineskip}%
  {\parindent\z@ \raggedright\normalfont
   \Huge\bfseries #1\par
   \nobreak
   \vspace{0.5\baselineskip}}}
\makeatother

% ----------------------------------------------------------------------------
% SECTION NUMBERING
% ----------------------------------------------------------------------------
\setcounter{secnumdepth}{1}
\renewcommand{\thesection}{\thechapter.\arabic{section}}

% Remove chapter numbers from TOC and chapter headings
\makeatletter
\renewcommand{\@chapapp}{}
\renewcommand{\thechapter}{}
\renewcommand{\thesection}{\arabic{chapter}.\arabic{section}}
\makeatother

% ----------------------------------------------------------------------------
% FIGURE AND TABLE NUMBERING
% ----------------------------------------------------------------------------
\counterwithout{figure}{chapter}
\counterwithout{table}{chapter}

\makeatletter
\renewcommand{\p@figure}{}
\renewcommand{\thefigure}{\arabic{figure}}
\renewcommand{\fnum@figure}{Figure~\arabic{figure}}
\makeatother
